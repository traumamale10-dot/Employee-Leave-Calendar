function initSigCanvas(){
  sigCanvas=document.getElementById('sigCanvas');if(!sigCanvas)return;
  const wrap=document.getElementById('sigWrap');
  sigCanvas.width=wrap.offsetWidth||520;
  sigCtx=sigCanvas.getContext('2d');
  sigCtx.strokeStyle=sigColor;sigCtx.lineWidth=2.5;sigCtx.lineCap='round';sigCtx.lineJoin='round';
  // Mouse
  sigCanvas.addEventListener('mousedown',e=>{isDrawing=true;const p=pos(e);sigCtx.beginPath();sigCtx.moveTo(p.x,p.y)});
  sigCanvas.addEventListener('mousemove',e=>{if(!isDrawing)return;const p=pos(e);sigCtx.lineTo(p.x,p.y);sigCtx.stroke()});
  sigCanvas.addEventListener('mouseup',()=>stopDraw());sigCanvas.addEventListener('mouseleave',()=>stopDraw());
  // Touch
  sigCanvas.addEventListener('touchstart',e=>{e.preventDefault();isDrawing=true;const p=pos(e.touches[0]);sigCtx.beginPath();sigCtx.moveTo(p.x,p.y)},{passive:false});
  sigCanvas.addEventListener('touchmove',e=>{e.preventDefault();if(!isDrawing)return;const p=pos(e.touches[0]);sigCtx.lineTo(p.x,p.y);sigCtx.stroke()},{passive:false});
  sigCanvas.addEventListener('touchend',()=>stopDraw());
}
function pos(e){const r=sigCanvas.getBoundingClientRect(),sx=sigCanvas.width/r.width,sy=sigCanvas.height/r.height;return{x:(e.clientX-r.left)*sx,y:(e.clientY-r.top)*sy}}
function stopDraw(){if(isDrawing){isDrawing=false;document.getElementById('sigPreview').classList.remove('hidden')}}
function clearSig(){if(sigCtx)sigCtx.clearRect(0,0,sigCanvas.width,sigCanvas.height);document.getElementById('sigPreview').classList.add('hidden')}
function setSigColor(c,el){sigColor=c;if(sigCtx)sigCtx.strokeStyle=c;document.querySelectorAll('.sig-color').forEach(x=>x.classList.remove('active'));el.classList.add('active')}
function isSigEmpty(){if(!sigCanvas)return true;const d=sigCtx.getImageData(0,0,sigCanvas.width,sigCanvas.height).data;return!d.some((v,i)=>i%4===3&&v>0)}
function getSigDataURL(){return sigCanvas?sigCanvas.toDataURL('image/png'):''}

function setPhoto(f){
  if(!f)return;if(f.size>10*1024*1024){showToast('ไฟล์ใหญ่เกิน 10 MB','error');return}
  const reader=new FileReader();
  reader.onload=ev=>{photoDataURL=ev.target.result;photoMime=f.type;document.getElementById('photoImg').src=photoDataURL;document.getElementById('photoPrev').classList.remove('hidden');document.getElementById('photoZone').style.display='none'};
  reader.readAsDataURL(f);
}

async function submitLearning(){
  const date=document.getElementById('logDate').value;
  const instructor=document.getElementById('logInstr').value;

function logCard(log, canDel=true){
  const d=log.date?new Date(log.date).toLocaleDateString('th-TH',{year:'numeric',month:'long',day:'numeric'}):'';
  const sigHtml=log.signatureUrl&&log.signatureUrl!=='(บันทึกแล้ว)'&&log.signatureUrl!==''
    ?`<div class="log-sig"><img src="${log.signatureUrl}" alt="ลายเซ็น" onerror="this.parentElement.style.display='none'"/></div>`
    :(log.signatureUrl==='(บันทึกแล้ว)'?`<span style="font-size:11px;color:var(--mint-500)">✅ มีลายเซ็น</span>`:'');
  const photoHtml=log.photoUrl&&log.photoUrl.startsWith('data:')?`<img src="${log.photoUrl}" class="log-photo-thumb" onclick="viewPhoto('${log.id}')" alt="รูปภาพ"/>`
    :log.photoUrl&&log.photoUrl!==''?`<a href="${log.photoUrl}" target="_blank"><img src="${log.photoUrl}" class="log-photo-thumb" onerror="this.style.display='none'" alt="รูป"/></a>`:'';
  const delBtn=canDel?`<button class="log-del" onclick="deleteLog('${log.id}')" title="ลบ">🗑</button>`:'';
  return`<div class="log-card" onclick="viewLog('${log.id}')">
    ${delBtn}
    <div class="log-date">📅 ${d}</div>
    <div class="log-topic">${log.topic}</div>
    <div class="log-instr">👩‍🏫 ${log.instructor||''}</div>
    ${log.notes?`<div class="log-notes">${log.notes}</div>`:''}
    <div class="log-media">${sigHtml}${photoHtml}</div>
  </div>`;
}
